    js
    //自定义tab绑定方法
    zui.tabcut({
        //父标签class或id
        wrap:'.mytab', 
        //菜单class或id
        menu:'.mytab-menu', 
        //内容class或id
        main:'.mytab-main',
        //切换方式，click或hover，不写此属性默认click
        target:'hover' 
    });

    css
    //开启的menu子类和main子类都会添加.active
    .mytab-menu{ margin-bottom: 10px;}  
    .mytab-menu span{ cursor: pointer;}
    .mytab-menu .active{ color: #55BBEE;}
    .mytab-main div{ display: none;}
    .mytab-main div.active{ display: block;}

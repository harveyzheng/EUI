    el.each((i,n)=>{
        let $n=$(n);
        // 初始化内容：替换<>，分割成数组
        let str=$n.html().replace(//g, ">").split("\n");
        let dom='
    ';
        let sp=0;
        str.forEach((s,i)=>{
            // 空格数量
            let spa=s.split(' ').length-1;
            // 记录开头空格数量
            if(i<1){
                sp=spa;
                $n.html('
    '+s.trim()+'
    ')
            }else{
                // 如果开头空格大于等于第一行，截取空格后的内容（删减空格）
                if(spa>=sp) s=s.substring(sp,s.length-1);
                if(i!=str.length-1) dom+='
  1. '+s+'
    2. ';
            };
        });
        dom+='
';
        $n.append(dom);
    });
